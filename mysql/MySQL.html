MySQL

MySQL은 데이터베이스 소프트웨어입니다. 일반적으로 데이터를 추가하거나 검색, 추출하는 기능을 모두 포함해서 데이터베이스라고 부릅니다.

MySQL은 세계에서 가장 많이 쓰이는 오픈 소스의 관계형 데이터베이스 관리 시스템(RDBMS)입니다. MySQL은 PHP 스크립트 언어화 상호 연동이 잘 되면서 오프소스로 개발된 무료 프로그램입니다. 그래서 홈페이지나 쇼핑몰(워드프레스, cafe24, 제로보드, 그누보드)등 일반적으로 웹 개발에 널리 사용하고 있습니다.

테이블

테이블 만들기

create table 테이블 이름; 
CREATE TABLE myMember (
    myMemberID int(10) unsigned auto_increment,
    youEmail varchar(40) NOT NULL,
    youName varchar(12) NOT NULL,
    youPass varchar(20) NOT NULL,
    youBirth varchar(12) NOT NULL,
    youAge varchar(10) NOT NULL,
    regTime int(11) NOT NULL,
    PRIMARY KEY (myMemberID)
) charset=utf8;

테이블 삭제

drop table 삭제할 테이블 이름 
mysql> drop table myMember;
Query OK, 0 rows affected (0.01 sec)

테이블 복사

전체 테이블(구조,데이터) 복사

create table 새로운 테이블명 select * from 복사할 테이블명; 
create table myMember1 select * from myMember;

테이블 구조만 복사

create table 새로운 테이블명 LIKE 복사할 테이블명; 
create table myMember1 LIKE myMember;

테이블 데이터만 복사

INSERT INTO 기본 테이블 SELECT * FROM 복사할 테이블; 
INSERT INTO myMember1 SELECT * FROM myMember;

테이블 수정

컬럼 추가하기

ALTER TABLE 테이블명 ADD 추가할 필드명 AFTER 필드명 위치; 
ALTER TABLE myData ADD youAddress varchar(40) AFTER youAge;

컬럼 수정하기

ALTER TABLE 테이블명 MODIFY 수정할 필드명

필드 컬럼 변경하기

ALTER TABLE myMember CHANGE youBirth youMonth int(10);;

필드 타입 변경하기

ALTER TABLE myMember MODIFY youName int(20);

컬럼 삭제하기

ALTER TABLE 테이블명 DROP 삭제할 필드명 
ALTER TABLE myMember DROP youPass;

테이블 데이터

데이터 입력하기

입력

INSERT INTO 테이블 이름(필드명) VALUES(데이터) 
INSERT INTO myMember(youEmail, youName, youPass, youBirth, youAge, regTime) VALUES('webstoryboy@naver.com','황상연','1234','1992-04-05','33','1234');

다중 입력

INSERT INTO 테이블 이름(필드명) VALUES(데이터),(데이터),(데이터)... 
INSERT INTO myMember(youEmail, youName, youPass, youBirth, youAge, regTime) VALUES('webstoryboy@naver.com','황상연','1234','1992-04-05','33','1234'), ('webstoryboy@naver.com','황상연','1234','1992-04-05','33','1234');

데이터 불러오기

전체 데이터 불러오기

SELECT * FROM 테이블 명; 
mysql> SELECT * FROM mymember;

조건 데이터 불러오기

SELECT * FROM 테이블 명 WHERE 조건;

myMemberID가 6번인 경우

SELECT * FROM mymember WHERE myMemberID = 6

myMemberID가 3번아 아닌 경우

SELECT * FROM mymember WHERE myMemberID <>3

멤버 아이디 조건 검색

SELECT * FROM mymember WHERE myMemberID >=2 AND myMemberID <=5;

범위 조건 검색

SELECT * FROM 테이블 명 WHERE 컬럼명 BETWEEN 시작 값 AND 끝 값; 
SELECT * FROM mymember WHERE myMemberID between 1 and 2;

범위 조건 반대 검색

SELECT * FROM mymember WHERE myMemberID not between 1 and 2;

해당 조건 검색

SELECT * FROM 테이블 명 WHERE 컬럼명 IN(값1, 값2.....) 
SELECT * FROM mymember WHERE myMemberID IN(1,3);

패턴 검색

SELECT * FROM 테이블 명 WHERE 컬럼명 LIKE

email 중에 naver 텍스트를 포함하는 경우

SELECT * FROM mymember WHERE youEmail LIKE '%naver%';

email 중에 webstoryboy 텍스트로 시작하는 경우

SELECT * FROM mymember WHERE youEmail LIKE 'webstoryboy%';

email 중에 com 텍스트로 끝나는 경우

SELECT * FROM mymember WHERE youEmail LIKE '%com';

멤버 이름 중에 첫글자가 '김'씨 이거나 '황'씨 인 경우를 검색

SELECT * FROM mymember WHERE youName LIKE '김%' OR youName LIKE '황%';

_밑줄, 글자에서 검색 (%문자열 _ 단일 1개의 문자)

SELECT * FROM myMember WHERE youName LIKE '_상_';
SELECT * FROM myMember WHERE youName LIKE '__연';

NULL

SELECT * FROM 테이블 명 WHERE 컬럼명 IS NULL 
SELECT * FROM myMember WHERE youName IS NULL;
SELECT * FROM myMember WHERE youName IS NOT NULL;

논리 연산자

SELECT * FROM 테이블 명 WHERE 컬럼명 >=2 AND 컬럼명 = 33; 
SELECT * FROM myMember WHERE myMemberID >=2 AND youAge = 33;
SELECT * FROM myMember WHERE myMemberID >=2 OR youAge = 33;
SELECT * FROM myMember WHERE (myMemberID=2 or myMemberID=3) AND youAge = 33;

데이터 출력 수

SELECT 컬럼명... FROM 테이블명 LIMIT 표시할 수 
SELECT youEmail from myMember LIMIT 2;
SELECT youEmail from myMember LIMIT 2, 3;
SELECT * from myMember LIMIT 2, 3;

데이터 정렬하기

내림차순/오름차순

SELECT * FROM 테이블명 ORDER by 컬럼명 정렬방식 
SELECT * from myMember ORDER by myMemberID desc;
SELECT * from myMember ORDER by myMemberID asc;

그룹정렬

SELECT * FROM 테이블명 GROUP by 컬럼명 
SELECT * from myMember GROUP by youName;
SELECT youName, count(*) from myMember GROUP by youName;

HAVING 추가 조건 설정

SELECT * FROM 테이블명 GROUP by 컬럼명 HAVING 조건

데이터 수정하기

UPDATE 테이블(필드명) SET 필드명 조건

모든 회원 비밀번호를 '1234' 변경

UPDATE mymember SET youPass = "1234";

회원 아이디 1번 이름 변경

UPDATE mymember SET youName = "강동재" WHERE myMemberID = 1;

정렬 수정

UPDATE myMember SET youName = "웹쓰" ORDER by myMemberID DESC LIMIT 3;

데이터 삭제하기

모든 데이터 삭제

DELETE FORM 테이블명
DELETE * FORM 테이블명 
DELETE FROM myMember1;

조건 데이터 삭제

DELETE FORM 테이블명 조건 = 필드명

회원 아이디 3번 레코드 삭제

DELETE FROM myMember WHERE myMemberID = 3;

조건 삭제

DELETE FROM myMember WHERE youPass = 124;

정렬 삭제

DELETE FROM myMember ORDER BY youName LIMIT 2;

회원 아이디 4,5,6번 레코드 삭제

mysql> DELETE FROM mymember WHERE myMemberID IN(4,5,6);
