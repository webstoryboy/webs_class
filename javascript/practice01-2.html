1. 구구단 출력하기

1.1 for문을 이용하여 2단부터 9단까지 출력하시오!

{
    for(let i=2; i<=9; i++){
        document.write("<br>");

        for(let j=1; j<=9; j++){
            let sum = i * j;
            
            document.write( i + " * " + j + " = " + sum, "<br>");
        }
    }
}

1.2 while문을 이용하여 2단부터 9단까지 출력하시오!

{
    let num1 = 2;
    let num2 = 1;

    while (num1 <= 9) {
        num2 = 1;
        while (num2 <= 9) {
            document.write(num1 + " * " + num2 + " = " + (num1 * num2) + "
");
            num2++;
        }
        num1++;
        document.write("<br>");
    }
}

1.3 사용자가 원하는 단을 받아 출력하시오!

조건 : 사용자가 원하는 단은 prompt()를 사용합니다. 클릭

{
    let num, result;

    document.querySelector(".btn03").addEventListener("click", function(){
        num = prompt("출력할 구구단 숫자를 입력하세요!", "2");

        for (let i=2; i<=9; i++) {
            result = num + " * " + i + " = " + (num * i) + "<br>";
            document.querySelector(".result03").innerHTML += result;
        }
    })
}
